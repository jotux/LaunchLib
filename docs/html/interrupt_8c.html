Attach function to GPIO pin edge interrupt.

To attach a function to ann interrupt we put it into a table of function pointers of which the port corresponds to the p1 or p2 array and the pin corresponds to the index into that array.

Parameters
[in]portGPIO port selection
[in]pinGPIO pin selection
[in]funcfunction pointer to attach
[in]typerising or falling edge
{
#define ATTACH_CASE(x) \
case x: \
p##x##_int_table[pin] = func; /* save the function pointer */ \
P##x##IFG &= ~_BV(pin); \
P##x##IE |= _BV(pin); /* interrupt enabled */ \
if (type == FALLING) \
{ \
P##x##IES |= _BV(pin); \
} \
else \
{ \
P##x##IES &= ~_BV(pin); \
} \
break;
switch (port)
{
ATTACH_CASE(1);
ATTACH_CASE(2);
#ifdef __MSP430FR5739__
ATTACH_CASE(3);
ATTACH_CASE(4);
#endif
}
}

void InterruptRunOnPort ( uint8_t  port)

Search interrupt function pointer lists and run functions corresponding to the interrupt that fired.

Go through each bit in the interrupt flag to see if it is set. If it is run the corresponding function.

Parameters
[in]portport to search
{
#define PORT_CASE(x) \
case x: \
{ \
uint8_t i = 0; \
for (i = 0;i < NUM_P##x##_INTS;i++) \
{ \
if ((P##x##IFG & _BV(i)) && p##x##_int_table[i] != NULL)\
{ \
p##x##_int_table[i](); \
P##x##IFG &= ~_BV(i); \
} \
} \
} \
break;
_DINT();
switch(port)
{
PORT_CASE(1);
PORT_CASE(2);
#ifdef __MSP430FR5739__
PORT_CASE(3);
PORT_CASE(4);
#endif
}
_EINT();
}
